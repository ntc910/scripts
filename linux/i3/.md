## Status bar
sudo pacman -S polybar ttf-nerd-fonts-symbols ttf-font-awesome

mkdir -p ~/.config/polybar

cp /usr/share/doc/polybar/config.ini ~/.config/polybar/config.ini

micro ~/.config/polybar/launch.sh

```sh
#!/usr/bin/env bash

# Tắt các tiến trình polybar đang chạy hiện tại (nếu có)
killall -q polybar

# Chờ cho đến khi các tiến trình cũ bị tắt hoàn toàn
while pgrep -u $UID -x polybar >/dev/null; do sleep 1; done

# Khởi chạy thanh bar có tên là "example" (tên mặc định trong file config.ini)
polybar example 2>&1 | tee -a /tmp/polybar.log & disown

echo "Polybar đã được kích hoạt..."
```

```sh
chmod +x ~/.config/polybar/launch.sh

mv ~/.config/polybar/launch.sh ~/repos/scripts/linux/i3/polybar_launch.sh
ln -s ~/repos/scripts/linux/i3/polybar_launch.sh ~/.config/polybar/launch.sh
```



- Modify config i3 file

```diff
--bar {
--        status_command i3status
--}
++exec_always --no-startup-id ~/.config/polybar/launch.sh
```

Reload with Mod Shift R

Backup
